CREATE TABLE LINEA
(
    ID_LINEA NUMBER NOT NULL,
    DESCRIPCION VARCHAR2(250) NOT NULL,
    CONSTRAINT PK_LINEA PRIMARY KEY (ID_LINEA)
 );
 
CREATE TABLE PRODUCTO
(
    ID_PRODUCTO NUMBER,
    ID_LINEA  NUMBER NOT NULL, 
    DESCRIPCION VARCHAR2(250) NOT NULL,
    PRECIO_COMPRA NUMBER(9,2) NOT NULL,
    PRECIO_VENTA NUMBER(9,2) NOT NULL,
    STOCK  NUMBER NOT NULL,
    CONSTRAINT PK_PRODUCTO PRIMARY KEY (ID_PRODUCTO),
    CONSTRAINT FK_PRODUCTO_LINEA FOREIGN KEY(ID_LINEA) REFERENCES  LINEA(ID_LINEA)
 );

CREATE TABLE EMPLEADO(
    ID_EMPLEADO NUMBER,
    ID_TIPOEMPLEADO NUMBER NOT NULL,
    NOMBRE VARCHAR2(255) NOT NULL,
    PATERNO VARCHAR2(255) NOT NULL,
    MATERNO VARCHAR2(255) NOT NULL,
     CONSTRAINT PK_EMPLEADO PRIMARY KEY (ID_EMPLEADO),
     CONSTRAINT FK_EMPLEADO_TIPOEMPLEADO FOREIGN KEY (ID_TIPOEMPLEADO) REFERENCES TIPOEMPLEADO(ID_TIPOEMPLEADO)
);

CREATE TABLE TIPOEMPLEADO(
    ID_TIPOEMPLEADO NUMBER,
    DESCRIPCION VARCHAR2(255) NOT NULL,
    CONSTRAINT PK_TIPOEMPLEADO PRIMARY KEY (ID_TIPOEMPLEADO)
);

CREATE TABLE CLIENTE(
    ID_CLIENTE NUMBER,
    RAZONSOCIAL VARCHAR2(255) NOT NULL,
    DIRECCION VARCHAR2(255) NOT NULL,
    RUCDNI VARCHAR(11) NOT NULL,
    TELEFONO VARCHAR(9) NOT NULL
);
CREATE TABLE VENTAPRESENCIAL(
    ID_VENTAPRESENCIAL NUMBER,
    ID_CLIENTE NUMBER NOT NULL,
    ID_EMPLEADO NUMBER NOT NULL,
    FECHA DEFAULT SYSDATE NOT NULL,
    MONTOTOTAL NUMBER(9,2) NOT NULL,
    CONSTRAINT PK_VENTAPRESENCIAL PRIMARY KEY (ID_VENTAPRESENCIAL),
);

CREATE TABLE DVENTAPRESENCIAL(
    ID_DVENTAPRESENCIAL NUMBER,
    ID_VENTA NUMBER NOT NULL,
    ID_PRODUCTO NUMBER NOT NULL,
    PRECIO NUMBER(9,2) NOT NULL,
    CANTIDAD NUMBER NOT NULL,
    IMPORTE NUMBER(9,2) NOT NULL
    CONSTRAINT PK_DVENTAPRESENCIAL PRIMARY KEY (DVENTAPRESENCIAL),
    CONSTRAINT FK_DVENTAPRESENCIAL_VENTA FOREIGN KEY (ID_VENTA) REFERENCES VENTA(ID_VENTA),
    CONSTRAINT FK_DVENTAPRESENCIAL_PRODUCTO  FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO(ID_PRODUCTO)
    
);

CREATE SEQUENCE SQ_LINEA;
CREATE SEQUENCE SQ_PRODUCTO START WITH 100;
CREATE SEQUENCE SQ_TIPOEMPLEADO;
CREATE SEQUENCE SQ_EMPLEADO;
CREATE SEQUENCE SQ_CLIENTE;
CREATE SEQUENCE SQ_VENTAPRESENCIAL;
CREATE SEQUENCE SQ_ DVENTAPRESENCIAL;
